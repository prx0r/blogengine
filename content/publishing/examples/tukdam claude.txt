        function getCurrentPhase() {

                }
                accumulatedTime += QUANTUM_PHASES[i].duration;
            }
            return { index: 0, progress: 0, phase: QUANTUM_PHASES[0] };
        }

                    oscillator: { type: "sine" },
                    envelope: { attack: 2.5, decay: 1.0, sustain: 0.4, release: 4.0 }
                });
                
                reverbSynth = new Tone.Synth({
                    oscillator: { type: "triangle" },
                    envelope: { attack: 3.0, decay: 1.5, sustain: 0.3, release: 5.0 }

                // Create quantum reverb space

                const delay = new Tone.PingPongDelay("4n", 0.1).toDestination();

            document.getElementById('meditation-guide').style.opacity = '0';
            
            draw();
