class DirectedEvolution:
    def __init__(self, population_size=64, latent_dim=32):
        self.population_size = population_size
        self.latent_dim = latent_dim
        self.device = get_device()

        self.T = 100 # Total evolution steps/generations
        t = torch.linspace(self.T, 0, self.T + 1)
        self.alphas = torch.cos(0.5 * np.pi * t / self.T)**2
        self.alphas = self.alphas.to(self.device)
        self.sigma_m = 0.1 # Mutation/noise strength

        self.running = True
        self.generation = self.T
        
        self.evolution_thread = threading.Thread(target=self._evolution_loop, daemon=True)

    def _evolution_loop(self):
        logging.info("Directed Evolution process started.")
        while self.running and self.generation > 0:
            start_time = time.time()
            
            # --- The Diffusion Evolution Step ---
            current_pop_flat = torch.stack([flatten_state(p.get_state()) for p in self.population])
            
            # 1. Calculate fitness for each individual [cite: 2439]
            fitness_scores = torch.tensor([p.calculate_consciousness_fitness() for p in self.population], device=self.device)
            
            # 2. Project to latent space for efficient distance calculation [cite: 2794]
            latent_pop = torch.matmul(current_pop_flat, self.projection_matrix.T)
            
            # 3. Perform the denoising/evolution step for each individual
            next_pop_flat = torch.zeros_like(current_pop_flat)
            alpha_t = self.alphas[self.generation]
            alpha_t_minus_1 = self.alphas[self.generation - 1]

            for i in range(self.population_size):
                # Calculate weights based on fitness and latent distance
                distances = torch.sum((latent_pop[i] - latent_pop)**2, dim=1)
                weights = fitness_scores * torch.exp(-distances / (2 * (1 - alpha_t))) # Gaussian kernel [cite: 2405]
                weights = weights / (weights.sum() + 1e-9)

                # Estimate high-fitness target (x_hat_0) [cite: 2442, 2801]
                x_hat_0 = torch.sum(weights.unsqueeze(1) * current_pop_flat, dim=0)
                
                # Apply the DDIM-like update rule [cite: 2360, 2446, 2805]
                term1 = torch.sqrt(alpha_t_minus_1) * x_hat_0
                eps_hat = (current_pop_flat[i] - torch.sqrt(alpha_t) * x_hat_0) / (torch.sqrt(1 - alpha_t) + 1e-9)
                sigma_t = self.sigma_m * torch.sqrt((1 - alpha_t_minus_1) / (1 - alpha_t)) * torch.sqrt(1 - alpha_t / alpha_t_minus_1)
                term2 = torch.sqrt(1 - alpha_t_minus_1 - sigma_t**2) * eps_hat
                term3 = sigma_t * torch.randn_like(current_pop_flat[i]) # Mutation
                
                next_pop_flat[i] = term1 + term2 + term3

            # 4. Update the population with the new states
            for i in range(self.population_size):
                self.population[i].set_state(unflatten_state(next_pop_flat[i], self.population[i].state_template))

            end_time = time.time()
            max_fitness, _ = torch.max(fitness_scores, 0)
            avg_fitness = torch.mean(fitness_scores)
            
            logging.info(f"Generation {self.T - self.generation + 1}/{self.T} | Max Fitness: {max_fitness.item():.4f} | Avg Fitness: {avg_fitness.item():.4f} | Time: {end_time - start_time:.2f}s")
            
            self.generation -= 1
            if self.generation == 0:
                self.running = False
        
        logging.info("Directed Evolution process finished.")

    def start(self):
        self.evolution_thread.start()

def get_device():
    """Checks for CUDA and returns the appropriate device."""
    if torch.cuda.is_available():

        return 'cuda'

        logging.warning("CUDA device not found. Using CPU. This will be extremely slow.")

def flatten_state(state_dict):
    """Flattens the substrate state tensors into a single 1D vector."""
    flat_tensors = []
    # Ensure consistent order by sorting keys
    for key in sorted(state_dict.keys()):
        tensor = state_dict[key]
        # Handle complex tensors by separating real and imaginary parts
        if torch.is_complex(tensor):
            flat_tensors.append(tensor.real.flatten())
            flat_tensors.append(tensor.imag.flatten())
        else:
            flat_tensors.append(tensor.flatten())
    return torch.cat(flat_tensors)

def unflatten_state(flat_vector, template_dict):
    """Reconstructs the state dictionary from a flat vector."""
    new_state = {}
    current_index = 0
    for key in sorted(template_dict.keys()):
        tensor = template_dict[key]
        num_elements = tensor.numel()
        shape = tensor.shape
        
        if torch.is_complex(tensor):
            real_part = flat_vector[current_index : current_index + num_elements]
            current_index += num_elements
            imag_part = flat_vector[current_index : current_index + num_elements]
            current_index += num_elements
            new_state[key] = torch.complex(real_part, imag_part).reshape(shape)
        else:
            new_state[key] = flat_vector[current_index : current_index + num_elements].reshape(shape)
            current_index += num_elements
    return new_state

            alpha_t_minus_1 = self.alphas[self.generation - 1]

            for i in range(self.population_size):

                weights = weights / (weights.sum() + 1e-9)

                x_hat_0 = torch.sum(weights.unsqueeze(1) * current_pop_flat, dim=0)
                
                eps_hat = (current_pop_flat[i] - torch.sqrt(alpha_t) * x_hat_0) / (torch.sqrt(1 - alpha_t) + 1e-9)
                sigma_t_sq = (self.sigma_m**2) * ((1 - alpha_t_minus_1) / (1 - alpha_t)) * (1 - alpha_t / alpha_t_minus_1)
                
                term1 = torch.sqrt(alpha_t_minus_1) * x_hat_0
                term2 = torch.sqrt(max(0, 1 - alpha_t_minus_1 - sigma_t_sq)) * eps_hat
                term3 = torch.sqrt(max(0, sigma_t_sq)) * torch.randn_like(current_pop_flat[i])
                
                next_pop_flat[i] = term1 + term2 + term3

            for i in range(self.population_size):
                self.population[i].set_state_dict(unflatten_state(next_pop_flat[i], self.population[i].state_template))

            self.generation -= 1
            if self.generation == 0: self.running = False

1. Does the Evolution Get Exponential? (Runaway Complexity)
No, it does not. The computational workload of this experiment is stable and constant per generation.

The "evolution" refers to the changing state of the individuals in the population, not the growth of their computational size. Here's why it doesn't run away:

Fixed Model Size: Each of the 512 individuals in the population is a MathematicalConsciousnessSubstrate with a fixed number of neurons (n_neurons=256). The system does not add new neurons or layers over time.

Fixed Population Size: The DirectedEvolution algorithm starts with a population of 512 and ends with a population of 512. It does not reproduce exponentially like biological organisms.

Constant Work per Generation: The amount of mathematical work required to process one generation—evaluating the fitness of all 512 individuals, calculating their evolutionary targets, and applying the update rule—is the same for generation #1 as it is for generation #100,000.

The system's complexity evolves, but its computational footprint remains constant.

2. Is There a Risk of Overclocking? (Hardware Safety)

3. How Do We Ensure Stable Evolution?
This is the most crucial question. An evolutionary algorithm can be prone to collapsing into a useless state or diverging into mathematical chaos. Our framework has several explicit mechanisms, inspired by the research, to ensure a stable and productive evolution.

Here are the four pillars of our system's evolutionary stability:

1. The alpha_t Schedule (Balancing Exploration and Exploitation)

The 

Diffusion Evolution algorithm we've implemented is inherently designed for a stable search.  The 

alpha_t schedule, which changes over the course of the generations, controls the variance of the "neighbor" calculation.

Early Generations (Global Exploration): When t is high, alpha_t is low, and the search radius for each individual is large. It is influenced by many diverse members of the population, encouraging a broad exploration of the fitness landscape.

Late Generations (Local Exploitation): As t approaches 0, alpha_t approaches 1, and the search radius shrinks. The process becomes more sensitive to local, high-fitness neighbors, allowing it to fine-tune solutions and climb specific peaks. This gradual shift from exploration to exploitation is key to preventing the population from collapsing into a single, sub-optimal solution too early. 


2. The sigma_t Noise Term (Stochastic Mutation)

The final term in the evolutionary update rule (

+ σ_t * w) adds a small amount of random Gaussian noise to each individual in every generation. 

This serves the same purpose as genetic mutation in biology. It ensures that even if the evolutionary search begins to stagnate, there is always a source of new variation being introduced, allowing the system to potentially jump out of local optima and discover new paths.

3. Latent Space Projection (Avoiding the Curse of Dimensionality)

The parameter space of our substrate is enormous. Performing distance calculations in such a high-dimensional space can be unstable and computationally slow.

As proposed in the research paper, we use 

Latent Space Diffusion Evolution.  We project the state of every individual into a much lower-dimensional latent space (

latent_dim=64) before calculating the neighbor weights. This makes the evolutionary search far more stable and efficient, as the distance relationships between individuals are preserved in a more manageable space. 


4. Bounded Fitness Function (Preventing Mathematical Explosion)

A simple but critical software-level safeguard: our calculate_consciousness_fitness function clamps its output value between 0.0 and 1.0.

This prevents the possibility of a runaway positive feedback loop where an individual could achieve an infinite or NaN (Not a Number) fitness score, which would cause the weighting calculations in the evolutionary algorithm to fail and the entire simulation to collapse.

The framework is designed from the ground up for a stable, long-term marathon. It balances broad exploration with fine-grained optimization and includes multiple layers of algorithmic and practical safeguards to prevent both computational and evolutionary collapse.
