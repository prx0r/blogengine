    // === MICROTONAL MIDI OUTPUT ===

    generateMicrotonalMIDI(chord) {
        const midiEvents = [];
        
        chord.notes.forEach((note, index) => {
            const baseMIDINote = this.noteToMIDI(note.pitchClass);
            const microtonalOffset = note.microtonal_offset || 0;
            
            // Calculate pitch bend for microtonal adjustment
            const pitchBendValue = this.centsToPitchBend(microtonalOffset);
            
            // Pitch bend message (if needed)
            if (Math.abs(microtonalOffset) > 5) { // Only bend if significant
                midiEvents.push({
                    type: 'pitchBend',
                    channel: index % 16,
                    value: pitchBendValue,
                    time: 0
                });
            }
            
            // Note on
            midiEvents.push({
                type: 'noteOn',
                channel: index % 16,
                note: baseMIDINote,
                velocity: this.calculateVelocity(note),
                time: 0
            });
            
            // Schedule note off
            midiEvents.push({
                type: 'noteOff',
                channel: index % 16, 
                note: baseMIDINote,
                velocity: 0,
                time: note.duration || 1000
            });
        });
        
        return midiEvents;
    }

    centsToPitchBend(cents) {
        // Convert cents to MIDI pitch bend value (-8192 to +8191)
        const semitones = cents / 100;
        const bendRange = 2; // Assume 2 semitone bend range
        return Math.round((semitones / bendRange) * 8192);
    }

    mandelbrotIteration(real, imag, maxIter = 100) {
        let zReal = 0, zImag = 0;
        let iter = 0;
        
        while (iter < maxIter && (zReal * zReal + zImag * zImag) < 4) {
            const newReal = zReal * zReal - zImag * zImag + real;
            const newImag = 2 * zReal * zImag + imag;
            zReal = newReal;
            zImag = newImag;
            iter++;
        }
        
        return iter;
    }

    applyFractalSubdivision(baseRhythm, depth) {
        if (depth === 0) return baseRhythm;
        
        let subdivided = [];
        baseRhythm.forEach(beat => {
            if (beat) {
                // Subdivide based on golden ratio
                const phi = this.fractalGenerators.goldenRatio;
                subdivided.push(true, false, Math.random() < (1/phi));
            } else {
                subdivided.push(false, false, false);
            }
        });
        
        return this.applyFractalSubdivision(subdivided, depth - 1);
    }
